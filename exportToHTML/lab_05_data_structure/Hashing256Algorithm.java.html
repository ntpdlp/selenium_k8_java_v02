4    import java.nio.charset.StandardCharsets;
5    import java.security.MessageDigest;
6    import java.security.NoSuchAlgorithmException;
7    
8    //Some Java Hashing Algorithm 'Security Hashing Algorithm': SHA-1, SHA-256, SHA-512
9    class Hashing256Algorithm {
10       public static String sha256(String input){
11           //SHA-256 provided by MessageDigest class
12           try {
13               MessageDigest messageDigest = MessageDigest.getInstance("SHA-256");
14               byte[] hash = messageDigest.digest(input.getBytes(StandardCharsets.UTF_8));
15   
16               StringBuilder outputHex = new StringBuilder();
17               for(byte b:hash){
18                   String hexStr = Integer.toHexString(0xff & b);
19                   if (hexStr.length() == 1){
20                       outputHex.append('0');
21                   }
22                   outputHex.append(hexStr);
23               }
24               return outputHex.toString();
25           } catch (NoSuchAlgorithmException e) {
26               throw new RuntimeException(e);
27           }
28       }
29   
30   
31       public static void main(String[] args) {
32           System.out.println("SHA-256 Hash: " + sha256("Hello, world"));
33       }
34   
35   }
