3    import java.util.ArrayList;
4    import java.util.Collections;
5    import java.util.Iterator;
6    import java.util.List;

8    public class LearnArrayList {
9    
10       public static void main(String[] args) {
11           ArrayList<Integer> myNums = new ArrayList<>();
12   
13           //common methods: add(),get(),set(),remove(),size()
14           myNums.add(1);
15           myNums.add(5);
16           myNums.add(8);
17           myNums.add(9);
18           System.out.println("size at initial: " + myNums.size()); //4
19           System.out.println(myNums);
20           Integer myChoice = myNums.get(1); //get(index) >> 5
21           myNums.set(1, 50); //set(index,value)
22           System.out.println("update value at 1 position, old value: " + myChoice); //5
23           System.out.println(myNums);
24           myNums.remove(0); // remove(index)
25           System.out.println("size after removing first element: " + myNums.size()); //3
26           System.out.println(myNums);
27   
28           //3 ways to loop through an arrayList elements
29           // 01: by using loop with for
30           // 02: by using loop with for enhanced
31           // 03: iterator
32           for (int index = 0; index < myNums.size(); index++) {
33               System.out.print(myNums.get(index) + "\t");
34           }
35           System.out.println("\r\n------------------");
36   
37           for (Integer num : myNums) {
38               System.out.print(num + "\t");
39           }
40           System.out.println("\r\n------------------");
41   
42           Iterator<Integer> iterMyNums = myNums.iterator();
43           while (iterMyNums.hasNext()) {
44               System.out.print(iterMyNums.next() + "\t");
45           }
46           System.out.println("\r\n------------------");
47   
48           //sort
49           List<String> inputNames = List.of("Jamie", "Peggy", "Tom", "Vovo", "Mike");
50           List<String> names = new ArrayList<>(inputNames);
51           System.out.println("Original list: " + names);
52           Collections.sort(names);
53           System.out.println("After sorting: " + names);
54           Collections.sort(names, Collections.reverseOrder());
55           System.out.println("Reverse Order sorting: " + names);
56       }
