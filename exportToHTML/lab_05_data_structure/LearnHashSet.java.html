20   package lab_05_data_structure;
21   
22   import java.util.*;
23   
24   public class LearnHashSet {
25   
26       public static void main(String[] args) {
27           //Pre: provide input data
28           List<String> namesInputdata = List.of("Jamie","Peggy","Peggy","Tom","Mike","Adnan","Eman","Eman");
29   
30           //common methods basic HashSet no duplication
31           // add(),remove(),contains(),size(),clear()
32           Set<String> names = new HashSet<>(namesInputdata);
33           System.out.println(names);//[Jamie, Peggy, Mike, Tom, Eman, Adnan]
34           System.out.println("Size: " + names.size());//6
35           String isFound = names.contains("Vovo")?"found":"not found";
36           System.out.println("Finding 'Vovo' result? " + isFound);//not found
37   
38           //Want to see sorted result => use TreeSet
39           Set<String> orderedNames = new TreeSet<>(names);
40           System.out.println("A-Z: " + orderedNames);//[Adnan, Eman, Jamie, Mike, Peggy, Tom]
41   
42           //Want to insert faster O(1) => use LinkedHashSet
43           Set<String> updateNames = new LinkedHashSet<>(names);
44           updateNames.add("Coco");
45           updateNames.add("Ronadol");
46           System.out.println(updateNames);//[Jamie, Peggy, Mike, Tom, Eman, Adnan, Coco, Ronadol]
47       }
48   }
